63- (instancetype)initWithForegroundTimeout:(NSInteger)foregroundTimeout andBackgroundTimeout:(NSInteger)backgroundTimeout andTracker:(SPTracker *)tracker {
64 if (self = [super init]) {
65 _foregroundTimeout = foregroundTimeout * 1000;
66 _backgroundTimeout = backgroundTimeout * 1000;
67 _inBackground = NO;
68 _isNewSession = YES;
69 _eventIndex = 0;
70 self.tracker = tracker;
71 self.dataPersistence = [SPDataPersistence dataPersistenceForNamespace:tracker.trackerNamespace];
72 NSMutableDictionary *storedSessionDict = self.dataPersistence.session.mutableCopy;
73 _userId = [self retrieveUserIdWithSessionDict:storedSessionDict];
74 if (storedSessionDict && _userId) {
75 [storedSessionDict setObject:_userId forKey:kSPSessionUserId];
76 _state = [[SPSessionState alloc] initWithStoredState:storedSessionDict];
77 }
78 if (!_state) {
79 SPLogTrack(nil, @"No previous session info available");
80 }
81
82 // Start session check
83 self.lastSessionCheck = [SPUtilities getTimestamp];
84 [self startChecker];
85
86 // Trigger notification for view changes
87 #if SNOWPLOW_TARGET_IOS || SNOWPLOW_TARGET_TV
88 [[NSNotificationCenter defaultCenter] addObserver:self
89 selector:@selector(updateInBackground)
90 name:UIApplicationWillResignActiveNotification
91 object:nil];
92 [[NSNotificationCenter defaultCenter] addObserver:self
93 selector:@selector(updateInForeground)

95 object:nil];

97 }

124- (NSDictionary *) getSessionDictWithEventId:(NSString *)eventId eventTimestamp:(long long)eventTimestamp userAnonymisation:(BOOL)userAnonymisation {
125 NSMutableDictionary *context = nil;
126 @synchronized (self) {
127 if (_isSessionCheckerEnabled) {
128 if ([self shouldUpdateSession]) {
129 [self updateSessionWithEventId:eventId eventTimestamp:eventTimestamp];
130 if (self.onSessionStateUpdate) {
131 dispatch_async(dispatch_get_global_queue(DISPATCH_QUEUE_PRIORITY_DEFAULT, 0), ^{
132 self.onSessionStateUpdate(self.state);
133 });
134 }
135 }
136 self.lastSessionCheck = [SPUtilities getTimestamp];
137 }
138
139 _eventIndex += 1;
140
141 context = _state.sessionContext;
142 [context setObject:[NSNumber numberWithInteger:_eventIndex] forKey:kSPSessionEventIndex];
143 }
144
145 if (userAnonymisation) { // mask the user identifier
146 NSMutableDictionary *copy = [[NSMutableDictionary alloc] initWithDictionary:context];
147 [copy setValue:kSPSessionAnonymousUserId forKey:kSPSessionUserId];
148 [copy setValue:[NSNull null] forKey:kSPSessionPreviousId];
149 return copy;
150 } else {
151 return context;
152 }
153}

185- (NSString *)retrieveUserIdWithSessionDict:(NSDictionary *)sessionDict {
186 NSString *userId = [sessionDict sp_stringForKey:kSPSessionUserId defaultValue:nil] ?: [SPUtilities getUUIDString];
187 // Session_UserID is available only if the session context is enabled.
188 // In a future version we would like to make it available even if the session context is disabled.
189 // For this reason, we store the Session_UserID in a separate storage (decoupled by session values)
190 // calling it Installation_UserID in order to remark that it isn't related to the session context.
191 // Although, for legacy, we need to copy its value in the Session_UserID of the session context
192 // as the session context schema (and related data modelling) requires it.
193 // For further details: https://discourse.snowplow.io/t/rfc-mobile-trackers-v2-0
194 NSUserDefaults *userDefaults = [NSUserDefaults standardUserDefaults];
195 NSString *storedUserId = [userDefaults stringForKey:kSPInstallationUserId];
196 if (storedUserId) {
197 userId = storedUserId;
198 } else {
199 [userDefaults setObject:userId forKey:kSPInstallationUserId];
200 }
201 return userId;
202}

204- (BOOL)shouldUpdateSession {
205 if (_isNewSession) {
206 return YES;
207 }
208 long long lastAccess = self.lastSessionCheck.longLongValue;
209 long long now = [SPUtilities getTimestamp].longLongValue;
210 NSInteger timeout = _inBackground ? _backgroundTimeout : _foregroundTimeout;
211 return now < lastAccess || now - lastAccess > timeout;
212}

214- (void)updateSessionWithEventId:(NSString *)eventId eventTimestamp: (long long)eventTimestamp {
215 _isNewSession = NO;
216 NSInteger sessionIndex = (_state.sessionIndex ?: 0) + 1;
217 NSString *eventISOTimestamp = [SPUtilities timestampToISOString:eventTimestamp];
218 _state = [[SPSessionState alloc] initWithFirstEventId:eventId firstEventTimestamp:eventISOTimestamp currentSessionId:[SPUtilities getUUIDString] previousSessionId:_state.sessionId sessionIndex:sessionIndex userId:_userId storage:@"LOCAL_STORAGE"];
219 NSDictionary<NSString *,NSObject *> *sessionToPersist = _state.sessionContext;
220 // Remove previousSessionId if nil because dictionaries with nil values aren't plist serializable
221 // and can't be stored with SPDataPersistence.
222 if (!_state.previousSessionId) {
223 NSMutableDictionary<NSString *,NSObject *> *sessionCopy = [sessionToPersist mutableCopy];
224 [sessionCopy removeObjectForKey:kSPSessionPreviousId];
225 sessionToPersist = sessionCopy;
226 }
227 self.dataPersistence.session = sessionToPersist;
228 _eventIndex = 0;
229}
