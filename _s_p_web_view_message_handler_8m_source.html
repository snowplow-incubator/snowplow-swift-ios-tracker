41- (void)userContentController:(WKUserContentController *)userContentController
42 didReceiveScriptMessage:(WKScriptMessage *)message {
43 NSDictionary *event = message.body[@"event"];
44 NSArray<NSDictionary *> *context = message.body[@"context"];
45 NSArray<NSString *> *trackers = message.body[@"trackers"];
46 NSString *command = message.body[@"command"];
47
48 if ([command isEqual:@"trackSelfDescribingEvent"]) {
49 [self trackSelfDescribing:event withContext:context andTrackers:trackers];
50 } else if ([command isEqual: @"trackStructEvent"]) {
51 [self trackStructEvent:event withContext:context andTrackers:trackers];
52 } else if ([command isEqual: @"trackPageView"]) {
53 [self trackPageView:event withContext:context andTrackers:trackers];
54 } else if ([command isEqual: @"trackScreenView"]) {
55 [self trackScreenView:event withContext:context andTrackers:trackers];
56 }

119- (void)track:(SPEvent *)event withContext:(NSArray<NSDictionary *> *)context andTrackers:(NSArray<NSString *> *)trackers {
120 if (context) {
121 [event setContexts:[self parseContext:context]];
122 }
123 if (trackers.count > 0) {
124 for (NSString *namespace in trackers) {
125 id<SPTrackerController> tracker = [SPSnowplow trackerByNamespace:namespace];
126 if (tracker) {
127 [tracker track:event];
128 }
129 }
130 } else {
131 [[SPSnowplow defaultTracker] track:event];
132 }
133}

135- (NSMutableArray<SPSelfDescribingJson *> *) parseContext:(NSArray<NSDictionary *> *)context {
136 NSMutableArray<SPSelfDescribingJson *> *contextEntities = [[NSMutableArray alloc] init];
137
138 for (NSDictionary *entityJson in context) {
139 NSString *schema = [entityJson objectForKey:@"schema"];
140 NSDictionary *payload = [entityJson objectForKey:@"data"];
141
142 if (schema && payload) {
143 SPSelfDescribingJson *entity = [[SPSelfDescribingJson alloc] initWithSchema:schema andDictionary:payload];
144 [contextEntities addObject:entity];
145 }
146 }
147
148 return contextEntities;
149}
