36+ (void) load {
37 static dispatch_once_t onceToken;
38 dispatch_once(&onceToken, ^{
39 //* Then swizzle */
40 Class class = [self class];
41
42 SEL originalSelector = @selector(viewDidAppear:);
43 SEL swizzledSelector = @selector(SP_viewDidAppear:);
44
45 Method originalMethod = class_getInstanceMethod(class, originalSelector);
46 Method swizzledMethod = class_getInstanceMethod(class, swizzledSelector);
47
48 BOOL didAddMethod =
49 class_addMethod(class,
50 originalSelector,
51 method_getImplementation(swizzledMethod),
52 method_getTypeEncoding(swizzledMethod));
53
54 if (didAddMethod) {
55 class_replaceMethod(class,
56 swizzledSelector,
57 method_getImplementation(originalMethod),
58 method_getTypeEncoding(originalMethod));
59 } else {
60 method_exchangeImplementations(originalMethod, swizzledMethod);
61 }
62 });
63}

67- (void) SP_viewDidAppear:(BOOL)animated {
68 [self SP_viewDidAppear:animated];
69
70 NSBundle *bundle = [NSBundle bundleForClass:[self class]];
71 if (![bundle.bundlePath hasPrefix:[NSBundle mainBundle].bundlePath]) {
72 // Ignore view controllers that don't start with the main bundle path
73 return;
74 }
75
76 // Construct userInfo
77 NSMutableDictionary * userInfo = [[NSMutableDictionary alloc] init];
78 userInfo[@"viewControllerClassName"] = NSStringFromClass([self class]);
79 userInfo[@"topViewControllerClassName"] = NSStringFromClass([[self _SP_topViewController] class]);
80 // `name` is set to snowplowId class instance variable if it exists (hence no @"id" in userInfo)
81 userInfo[@"name"] = [self _SP_getViewControllerName];
82 userInfo[@"type"] = [[NSNumber alloc] initWithInteger:[self _SP_getTopViewControllerType]];
83
84 // Send notification to tracker
85 [[NSNotificationCenter defaultCenter]
86 postNotificationName:@"SPScreenViewDidAppear"
87 object:self
88 userInfo:userInfo];
89}

95- (NSString *) _SP_getSnowplowId {
96 NSString * propertyName = @"snowplowId";
97 SEL selector = NSSelectorFromString(propertyName);
98 BOOL propertyExists = [self respondsToSelector:selector];
99 if (propertyExists) {
100 id value = [self valueForKey:propertyName];
101 if (value && ([value isKindOfClass:[NSString class]] || [NSStringFromClass([value class]) isEqualToString:@"Swift.String"])) {
102 return value;
103 }
104 }
105 return nil;
106}

108- (NSString *) _SP_getViewControllerName {
109 NSString * viewControllerName = [self _SP_getViewControllerName:self];
110 NSString * topViewControllerName = [self _SP_getViewControllerName:[self _SP_topViewController]];
111
112 if ([self _SP_validateString:viewControllerName]) {
113 return viewControllerName;
114 } else if ([self _SP_validateString:topViewControllerName]) {
115 return topViewControllerName;
116 }
117
118 return @"Unknown";
119}

121- (NSString *) _SP_getViewControllerName:(UIViewController *)viewController {
122 // check if there's an instance variable snowplowId
123 NSString * viewControllerSnowplowId = [viewController _SP_getSnowplowId];
124 if ([self _SP_validateString:viewControllerSnowplowId]) {
125 return viewControllerSnowplowId;
126 }
127
128 // otherwise return the class name
129 NSString * viewControllerClassName = NSStringFromClass([viewController class]);
130 if ([self _SP_validateString:viewControllerClassName]) {
131 return viewControllerClassName;
132 }
133
134 return nil;

137- (SPScreenType) _SP_getViewControllerType:(UIViewController *)viewController {
138 if ([viewController isKindOfClass:[UINavigationController class]]) {
139 return SPScreenTypeNavigation;
140 }
141 if ([viewController isKindOfClass:[UITabBarController class]]) {
142 return SPScreenTypeTabBar;
143 }
144 if (viewController.presentedViewController) {
145 return SPScreenTypeModal;
146 }
147 if ([viewController isKindOfClass:[UIPageViewController class]]) {
148 return SPScreenTypePageView;
149 }
150 if ([viewController isKindOfClass:[UISplitViewController class]]) {
151 return SPScreenTypeSplitView;
152 }
153 Class uiPopoverPresentationControllerClass = NSClassFromString(@"UIPopoverPresentationController");
154 if (uiPopoverPresentationControllerClass && [viewController isKindOfClass:uiPopoverPresentationControllerClass]) {
155 return SPScreenTypePopoverPresentation;
156 }
157 return SPScreenTypeDefault;
158}

164- (UIViewController *) _SP_topViewController {
165 UIWindow *keyWindow = self.view.window;
166 if (!keyWindow) {
167 return nil;
168 }
169 return [self _SP_topViewController:keyWindow.rootViewController];
170}

172- (UIViewController *) _SP_topViewController:(UIViewController *)rootViewController {
173 if ([rootViewController isKindOfClass:[UINavigationController class]]) {
174 UINavigationController *navigationController = (UINavigationController *)rootViewController;
175 return [self _SP_topViewController:[navigationController.viewControllers lastObject]];
176 }
177
178 if ([rootViewController isKindOfClass:[UITabBarController class]]) {
179 UITabBarController *tabController = (UITabBarController *)rootViewController;
180 return [self _SP_topViewController:tabController.selectedViewController];
181 }
182
183 UIViewController *presentedViewController = rootViewController.presentedViewController;
184 if (presentedViewController != nil) {
185 return [self _SP_topViewController:presentedViewController];
186 }
187
188 return rootViewController;
189}
