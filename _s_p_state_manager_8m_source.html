49- (void)addOrReplaceStateMachine:(id<SPStateMachineProtocol>)stateMachine identifier:(NSString *)stateMachineIdentifier {
50 @synchronized (self) {
51 id<SPStateMachineProtocol> previousStateMachine = [self.identifierToStateMachine objectForKey:stateMachineIdentifier];
52 if (previousStateMachine) {
53 if ([stateMachine isMemberOfClass:[previousStateMachine class]]) {
54 return;
55 }
56 [self removeStateMachine:stateMachineIdentifier];
57 }
58 self.identifierToStateMachine[stateMachineIdentifier] = stateMachine;
59 [self.stateMachineToIdentifier setObject:stateMachineIdentifier forKey:stateMachine];
60 [self addToSchemaRegistry:self.eventSchemaToStateMachine
61 schemas:[stateMachine subscribedEventSchemasForTransitions]
62 stateMachine:stateMachine];
63 [self addToSchemaRegistry:self.eventSchemaToEntitiesGenerator
64 schemas:[stateMachine subscribedEventSchemasForEntitiesGeneration]
65 stateMachine:stateMachine];
66 [self addToSchemaRegistry:self.eventSchemaToPayloadUpdater
67 schemas:[stateMachine subscribedEventSchemasForPayloadUpdating]
68 stateMachine:stateMachine];
69 }
70}

72- (BOOL)removeStateMachine:(NSString *)stateMachineIdentifier {
73 id<SPStateMachineProtocol> stateMachine = self.identifierToStateMachine[stateMachineIdentifier];
74 if (!stateMachine) {
75 return NO;
76 }
77 [self.identifierToStateMachine removeObjectForKey:stateMachineIdentifier];
78 [self.stateMachineToIdentifier removeObjectForKey:stateMachine];
79 [self.trackerState removeStateWithIdentifier:stateMachineIdentifier];
80 [self removeFromSchemaRegistry:self.eventSchemaToStateMachine
81 schemas:[stateMachine subscribedEventSchemasForTransitions]
82 stateMachine:stateMachine];
83 [self removeFromSchemaRegistry:self.eventSchemaToEntitiesGenerator
84 schemas:[stateMachine subscribedEventSchemasForEntitiesGeneration]
85 stateMachine:stateMachine];
86 [self removeFromSchemaRegistry:self.eventSchemaToPayloadUpdater
87 schemas:[stateMachine subscribedEventSchemasForPayloadUpdating]
88 stateMachine:stateMachine];
89 return YES;
90}

92- (id<SPTrackerStateSnapshot>)trackerStateForProcessedEvent:(SPEvent *)event {
93 @synchronized (self) {
94 if ([event isKindOfClass:SPSelfDescribingAbstract.class]) {
95 SPSelfDescribingAbstract *sdEvent = (SPSelfDescribingAbstract *)event;
96 NSMutableArray<id<SPStateMachineProtocol>> *stateMachines = self.eventSchemaToStateMachine[sdEvent.schema].mutableCopy ?: [NSMutableArray new];
97 [stateMachines addObjectsFromArray:self.eventSchemaToStateMachine[@"*"]];
98 for (id<SPStateMachineProtocol> stateMachine in stateMachines) {
99 NSString *stateIdentifier = [self.stateMachineToIdentifier objectForKey:stateMachine];
100 SPStateFuture *previousStateFuture = [self.trackerState stateFutureWithIdentifier:stateIdentifier];
101 SPStateFuture *currentStateFuture = [[SPStateFuture alloc] initWithEvent:sdEvent previousState:previousStateFuture stateMachine:stateMachine];
102 [self.trackerState setStateFuture:currentStateFuture identifier:stateIdentifier];
103 // TODO: Remove early state computation.
104 /*
105 The early state-computation causes low performance as it's executed synchronously on
106 the track method thread. Ideally, the state computation should be executed only on
107 entities generation or payload updating (outputs). In that case there are two problems
108 to address:
109 - long chains of StateFuture filling the memory (in case the outputs are not generated)
110 - event object reuse by the user (the event object in the StateFuture could be modified
111 externally)
112 Remove the early state-computation only when these two problems are fixed.
113 */
114 [currentStateFuture state]; // Early state-computation
115 }
116 }
117 return self.trackerState.snapshot;
118 }

121- (NSArray<SPSelfDescribingJson *> *)entitiesForProcessedEvent:(id<SPInspectableEvent>)event {
122 @synchronized (self) {
123 NSMutableArray<SPSelfDescribingJson *> *result = [NSMutableArray new];
124 NSMutableArray<id<SPStateMachineProtocol>> *stateMachines = self.eventSchemaToEntitiesGenerator[event.schema].mutableCopy ?: [NSMutableArray new];
125 [stateMachines addObjectsFromArray:self.eventSchemaToEntitiesGenerator[@"*"]];
126 for (id<SPStateMachineProtocol> stateMachine in stateMachines) {
127 NSString *stateIdentifier = [self.stateMachineToIdentifier objectForKey:stateMachine];
128 id<SPState> state = [event.state stateWithIdentifier:stateIdentifier];
129 NSArray<SPSelfDescribingJson *> *entities = [stateMachine entitiesFromEvent:event state:state];
130 if (entities) {
131 [result addObjectsFromArray:entities];
132 }
133 }
134 return result;
135 }

138- (BOOL)addPayloadValuesToEvent:(id<SPInspectableEvent>)event {
139 @synchronized (self) {
140 int failures = 0;
141 NSMutableArray<id<SPStateMachineProtocol>> *stateMachines = self.eventSchemaToPayloadUpdater[event.schema] ?: [NSMutableArray new];
142 [stateMachines addObjectsFromArray:self.eventSchemaToPayloadUpdater[@"*"]];
143 for (id<SPStateMachineProtocol> stateMachine in stateMachines) {
144 NSString *stateIdentifier = [self.stateMachineToIdentifier objectForKey:stateMachine];
145 id<SPState> state = [event.state stateWithIdentifier:stateIdentifier];
146 NSDictionary<NSString *, NSObject *> *payloadValues = [stateMachine payloadValuesFromEvent:event state:state];
147 if (payloadValues && ![event addPayloadValues:payloadValues]) {
148 failures++;
149 }
150 }
151 return failures == 0;
152 }

157- (void)addToSchemaRegistry:(NSMutableDictionary<NSString *, NSMutableArray<id<SPStateMachineProtocol>> *> *)schemaRegistry schemas:(NSArray<NSString *> *)schemas stateMachine:(id<SPStateMachineProtocol>)stateMachine {
158 for (NSString *eventSchema in schemas) {
159 NSMutableArray *array = schemaRegistry[eventSchema];
160 if (!array) {
161 array = [NSMutableArray new];
162 schemaRegistry[eventSchema] = array;
163 }
164 [array addObject:stateMachine];
165 }

168- (void)removeFromSchemaRegistry:(NSMutableDictionary<NSString *, NSMutableArray<id<SPStateMachineProtocol>> *> *)schemaRegistry schemas:(NSArray<NSString *> *)schemas stateMachine:(id<SPStateMachineProtocol>)stateMachine {
169 for (NSString *eventSchema in schemas) {
170 NSMutableArray *array = schemaRegistry[eventSchema];
171 [array removeObject:stateMachine];
