89- (NSDictionary<NSString *, NSDictionary<NSString *, NSObject *> *> *)data {
90 @synchronized (self) {
91 if (!self.isStoredOnFile) {
92 return [[NSUserDefaults standardUserDefaults] dictionaryForKey:self.userDefaultsKey] ?: @{};
93 }
94 NSMutableDictionary<NSString *, NSDictionary<NSString *, NSObject *> *> *result =
95 [NSMutableDictionary dictionaryWithContentsOfURL:self.fileUrl];
96
97 if (!result) {
98 // Initialise
99 result = [NSMutableDictionary new];
100 // Migrate legacy session data
101 NSMutableDictionary *sessionDict = [self sessionDictionaryFromLegacyTrackerV2_2].mutableCopy
102 ?: [self sessionDictionaryFromLegacyTrackerV1].mutableCopy
103 ?: [NSMutableDictionary new];
104 // Add missing fields
105 [sessionDict setObject:@"" forKey:kSPSessionFirstEventId];
106 [sessionDict setObject:@"LOCAL_STORAGE" forKey:kSPSessionStorage];
107 // Wrap up
108 [result setObject:sessionDict forKey:sessionKey];
109 [self storeDictionary:result fileURL:self.fileUrl];
110 }
111
112 return result;
113 }

173- (NSURL *)createDirectoryUrl {
174 NSFileManager *fileManager = [NSFileManager defaultManager];
175 NSURL *url = [fileManager URLsForDirectory:NSApplicationSupportDirectory inDomains:NSUserDomainMask].lastObject;
176 url = [url URLByAppendingPathComponent:@"snowplow" isDirectory:YES];
177 NSError *error = nil;
178 if ([url checkResourceIsReachableAndReturnError:&error]) {
179 return url;
180 }
181 if ([fileManager createDirectoryAtURL:url withIntermediateDirectories:YES attributes:nil error:&error]) {
182 return url;
183 }
184 SPLogError(@"Unable to create directory for tracker data persistence: %@", error.localizedDescription);
185 return nil;
186}

188- (BOOL)storeDictionary:(NSDictionary *)dictionary fileURL:(NSURL *)fileUrl {
189 BOOL result = NO;
190 NSError *error = nil;
191 if (@available(iOS 11.0, macOS 10.13, watchOS 4.0, *)) {
192 result = [dictionary writeToURL:fileUrl error:&error];
193 } else {
194 result = [dictionary writeToURL:fileUrl atomically:YES];
195 }
196 if (result) {
197 return YES;
198 }
199 SPLogError(@"Unable to write file for sessions: %@", error.localizedDescription ?: @"-");
200 return NO;
201}
