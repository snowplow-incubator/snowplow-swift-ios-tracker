77+ (NSString *)stringFromNotificationTrigger:(nullable UNNotificationTrigger *)trigger API_AVAILABLE(ios(10.0)) {
78 NSMutableString * triggerType = [[NSMutableString alloc] initWithString:@"UNKNOWN"];
79 NSString * triggerClass = NSStringFromClass([trigger class]);
80 if ([triggerClass isEqualToString:@"UNTimeIntervalNotificationTrigger"]) {
81 [triggerType setString:@"TIME_INTERVAL"];
82 } else if ([triggerClass isEqualToString:@"UNCalendarNotificationTrigger"]) {
83 [triggerType setString:@"CALENDAR"];
84 } else if ([triggerClass isEqualToString:@"UNLocationNotificationTrigger"]) {
85 [triggerType setString:@"LOCATION"];
86 } else if ([triggerClass isEqualToString:@"UNPushNotificationTrigger"]) {
87 [triggerType setString:@"PUSH"];
88 }
89 return (NSString *)triggerType;
90}

167 // modify contentAvailable value "1" and "0" to @YES and @NO to comply with schema
168 if (![[_userInfo valueForKeyPath:@"aps.contentAvailable"] isEqual:nil] &&
169 [[_userInfo objectForKey:@"aps"] isKindOfClass:[NSDictionary class]]) {
170 aps = [[NSMutableDictionary alloc] initWithDictionary:_userInfo[@"aps"]];
171
172 if ([[_userInfo valueForKeyPath:@"aps.contentAvailable"] isEqual:@1]) {

175 [aps setObject:@NO forKey:@"contentAvailable"];

178 [newUserInfo setObject:aps forKey:@"aps"];

181 }

183 NSMutableArray<NSDictionary *> * converting = [[NSMutableArray alloc] init];

188 newAttachment[kSPPnAttachmentType] = [attachment valueForKey:@"type"];
189 [converting addObject: (NSDictionary *)[newAttachment copy]];
