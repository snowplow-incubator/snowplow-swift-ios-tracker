47- (instancetype)initWithPlatformContext:(BOOL)platformContext geoLocationContext:(BOOL)geoContext subjectConfiguration:(SPSubjectConfiguration *)config {
48 if (self = [super init]) {
49 self.platformContext = platformContext;
50 self.geoLocationContext = geoContext;
51 _standardDict = [[SPPayload alloc] init];
52 _platformContextManager = [[SPPlatformContext alloc] init];
53 [self setStandardDict];
54 [self setGeoDict];
55 if (config) {
56 if (config.userId) [self setUserId:config.userId];
57 if (config.networkUserId) [self setNetworkUserId:config.networkUserId];
58 if (config.domainUserId) [self setDomainUserId:config.domainUserId];
59 if (config.useragent) [self setUseragent:config.useragent];
60 if (config.ipAddress) [self setIpAddress:config.ipAddress];
61 NSString *timezone = config.timezone ?: [NSTimeZone localTimeZone].name;
62 if (timezone) [self setTimezone:timezone];
63 NSString *language = config.language ?: [NSLocale preferredLanguages].firstObject;
64 if (config.language) [self setLanguage:language];
65 if (config.screenResolution) {
66 SPSize *size = config.screenResolution;
67 [self setResolutionWithWidth:size.width andHeight:size.height];
68 }
69 if (config.screenViewPort) {
70 SPSize *size = config.screenViewPort;
71 [self setViewPortWithWidth:size.width andHeight:size.height];
72 }
73 if (config.colorDepth) {
74 [self setColorDepth:config.colorDepth.integerValue];
75 }
76 // geolocation
77 if (config.geoLatitude) {
78 [self setGeoLatitude:config.geoLatitude.floatValue];
79 }
80 if (config.geoLongitude) {
81 [self setGeoLongitude:config.geoLongitude.floatValue];
82 }
83 if (config.geoLatitudeLongitudeAccuracy) {
84 [self setGeoLatitudeLongitudeAccuracy:config.geoLatitudeLongitudeAccuracy.floatValue];
85 }
86 if (config.geoAltitude) {
87 [self setGeoAltitude:config.geoAltitude.floatValue];
88 }
89 if (config.geoAltitudeAccuracy) {
90 [self setGeoAltitudeAccuracy:config.geoAltitudeAccuracy.floatValue];
91 }
92 if (config.geoSpeed) {
93 [self setGeoSpeed:config.geoSpeed.floatValue];
94 }
95 if (config.geoBearing) {
96 [self setGeoBearing:config.geoBearing.floatValue];
97 }
98 if (config.geoTimestamp) {
99 [self setGeoTimestamp:config.geoTimestamp];
100 }
101 }
102 }
103 return self;
104}

108- (SPPayload *) getStandardDictWithUserAnonymisation:(BOOL)userAnonymisation {
109 if (userAnonymisation) {
110 NSMutableDictionary *copy = [[NSMutableDictionary alloc] initWithDictionary:[_standardDict getAsDictionary]];
111 [copy removeObjectForKey:kSPUid];
112 [copy removeObjectForKey:kSPDomainUid];
113 [copy removeObjectForKey:kSPNetworkUid];
114 [copy removeObjectForKey:kSPIpAddress];
115 return [[SPPayload alloc] initWithNSDictionary:copy];
116 }
117 return _standardDict;
118}
