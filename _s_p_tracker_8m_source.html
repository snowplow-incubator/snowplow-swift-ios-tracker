80void uncaughtExceptionHandler(NSException *exception) {
81 NSArray* symbols = [exception callStackSymbols];
82 NSString * stacktrace = [NSString stringWithFormat:@"Stacktrace:\n%@", symbols];
83 NSString * message = [exception reason];
84 dispatch_sync(dispatch_get_global_queue(DISPATCH_QUEUE_PRIORITY_DEFAULT, 0), ^{
85 if (message == nil || [message length] == 0) {
86 return;
87 }
88
89 // Construct userInfo
90 NSMutableDictionary<NSString *, NSObject *> *userInfo = [NSMutableDictionary new];
91 userInfo[@"message"] = message;
92 userInfo[@"stacktrace"] = stacktrace;
93
94 // Send notification to tracker
95 [[NSNotificationCenter defaultCenter]
96 postNotificationName:@"SPCrashReporting"
97 object:nil
98 userInfo:userInfo];
99
100 [NSThread sleepForTimeInterval:2.0f];
101 });
102}

243- (void)checkInstall {
244 if (_installEvent) {
245 SPInstallTracker * installTracker = [[SPInstallTracker alloc] init];
246 NSDate *previousTimestamp = installTracker.previousInstallTimestamp;
247 [installTracker clearPreviousInstallTimestamp];
248 if (!installTracker.isNewInstall && previousTimestamp == nil) {
249 return;
250 }
251 SPSelfDescribingJson *installEvent = [[SPSelfDescribingJson alloc] initWithSchema:kSPApplicationInstallSchema andData:@{}];
252 SPSelfDescribing *event = [[SPSelfDescribing alloc] initWithEventData:installEvent];
253 event.trueTimestamp = previousTimestamp; // it can be nil
254 [self track:event];
255 }

258- (void) setTrackerData {
259 NSString *trackerVersion = kSPVersion;
260 if ([_trackerVersionSuffix length]) {
261 NSMutableCharacterSet *allowedCharSet = [NSMutableCharacterSet alphanumericCharacterSet];
262 [allowedCharSet formUnionWithCharacterSet:[NSCharacterSet characterSetWithCharactersInString:@".-"]];
263 NSString *suffix = [[_trackerVersionSuffix componentsSeparatedByCharactersInSet:[allowedCharSet invertedSet]] componentsJoinedByString:@""];
264 if ([suffix length]) {
265 trackerVersion = [NSString stringWithFormat:@"%@ %@", trackerVersion, suffix];
266 }
267 }
268 _trackerData = [NSMutableDictionary dictionaryWithObjectsAndKeys:
269 trackerVersion, kSPTrackerVersion,
270 _trackerNamespace, kSPNamespace,
271 _appId != nil ? _appId : [NSNull null], kSPAppId, nil];
272}

501 NSString *type = stringWithSPScreenType([[[notification userInfo] objectForKey:@"type"] integerValue]);
502 NSString *topViewControllerClassName = [[notification userInfo] objectForKey:@"topViewControllerClassName"];

510 [self track:event];
511 }

541- (NSUUID *)track:(SPEvent *)event {
542 if (!event || !_dataCollection) return nil;
543 [event beginProcessingWithTracker:self];
544 NSUUID *eventId = [self processEvent:event];
545 [event endProcessingWithTracker:self];
546 return eventId;
547}

549#pragma mark - Event Decoration
550
551- (NSUUID *)processEvent:(SPEvent *)event {
552 SPTrackerState *stateSnapshot;
553 @synchronized (self) {
554 stateSnapshot = [self.stateManager trackerStateForProcessedEvent:event];
555 }
556 SPTrackerEvent *trackerEvent = [[SPTrackerEvent alloc] initWithEvent:event state:stateSnapshot];
557 [self transformEvent:trackerEvent];
558 SPPayload *payload = [self payloadWithEvent:trackerEvent];
559 [_emitter addPayloadToBuffer:payload];
560 return [trackerEvent eventId];
561}
562
563- (void)transformEvent:(SPTrackerEvent *)event {
564 // Application_install event needs the timestamp to the real installation event.
565 if ([event.schema isEqualToString:kSPApplicationInstallSchema] && event.trueTimestamp) {
566 event.timestamp = event.trueTimestamp.timeIntervalSince1970 * 1000;
567 event.trueTimestamp = nil;
568 }
569 // Payload can be optionally updated with values based on internal state
570 [self.stateManager addPayloadValuesToEvent:event];
571}
572
573- (SPPayload *)payloadWithEvent:(SPTrackerEvent *)event {
574 SPPayload *payload = [SPPayload new];
575 payload.allowDiagnostic = !event.isService;
576
577 [self addBasicPropertiesToPayload:payload event:event];
578 if (event.isPrimitive) {
579 [self addPrimitivePropertiesToPayload:payload event:event];
580 } else {
581 [self addSelfDescribingPropertiesToPayload:payload event:event];
582 }
583 NSMutableArray<SPSelfDescribingJson *> *contexts = event.contexts;
584 [self addBasicContextsToContexts:contexts event:event];
585 [self addGlobalContextsToContexts:contexts event:event];
586 [self addStateMachineEntitiesToContexts:contexts event:event];
587 [self wrapContexts:contexts toPayload:payload];
588 if (!event.isPrimitive) {
589 // TODO: To remove when Atomic table refactoring is finished
590 [self workaroundForCampaignAttributionEnrichment:payload event:event contexts:contexts];
591 }
592 return payload;
593}

595- (void)addBasicPropertiesToPayload:(SPPayload *)payload event:(SPTrackerEvent *)event {

628/*
629 This is needed because the campaign-attribution-enrichment (in the pipeline) is able to parse
630 the `url` and `referrer` only if they are part of a PageView event.
631 The PageView event is an atomic event but the DeepLinkReceived and ScreenView are SelfDescribing events.
632 For this reason we copy these two fields in the atomic fields in order to let the enrichment
633 to process correctly the fields even if the event is not a PageView and it's a SelfDescribing event.
634 This is a hack that should be removed once the atomic event table is dismissed and all the events
635 will be SelfDescribing.
636 */
637- (void)workaroundForCampaignAttributionEnrichment:(SPPayload *)payload event:(SPTrackerEvent *)event contexts:(NSMutableArray<SPSelfDescribingJson *> *)contexts {
638 NSString *url = nil;
639 NSString *referrer = nil;
640
641 if ([event.schema isEqualToString:kSPDeepLinkReceivedSchema]) {
642 url = (NSString *)[event.payload objectForKey:kSPDeepLinkReceivedParamUrl];
643 referrer = (NSString *)[event.payload objectForKey:kSPDeepLinkReceivedParamReferrer];
644 } else if ([event.schema isEqualToString:kSPScreenViewSchema]) {
645 for (SPSelfDescribingJson *entity in contexts) {
646 if ([[entity schema] isEqualToString:kSPDeepLinkSchema]) {
647 NSDictionary *data = (NSDictionary *)[entity data];
648 url = (NSString *)[data valueForKey:kSPDeepLinkReceivedParamUrl];
649 referrer = (NSString *)[data valueForKey:kSPDeepLinkParamReferrer];
650 break;
651 }
652 }
653 }
654
655 if (url != nil) {
656 [payload addValueToPayload:url forKey:kSPPageUrl];
657 }
658 if (referrer != nil) {
659 [payload addValueToPayload:referrer forKey:kSPPageRefr];
660 }
661}

667- (void)addBasicContextsToContexts:(NSMutableArray<SPSelfDescribingJson *> *)contexts eventId:(NSString *)eventId eventTimestamp:(long long)eventTimestamp isService:(BOOL)isService {
668 if (_subject) {
669 NSDictionary * platformDict = [[_subject getPlatformDictWithUserAnonymisation:self.userAnonymisation] getAsDictionary];
670 if (platformDict != nil) {
671 [contexts addObject:[[SPSelfDescribingJson alloc] initWithSchema:_platformContextSchema andData:platformDict]];
672 }
673 NSDictionary * geoLocationDict = [_subject getGeoLocationDict];
674 if (geoLocationDict != nil) {
675 [contexts addObject:[[SPSelfDescribingJson alloc] initWithSchema:kSPGeoContextSchema andData:geoLocationDict]];
676 }
677 }
678
679 if (_applicationContext) {
680 SPSelfDescribingJson * contextJson = [SPUtilities getApplicationContext];
681 if (contextJson != nil) {
682 [contexts addObject:contextJson];
683 }
684 }
685
686 if (isService) {
687 return;
688 }
689
690 // Add session
691 if (_session) {
692 NSDictionary *sessionDict = [_session getSessionDictWithEventId:eventId eventTimestamp:eventTimestamp userAnonymisation:self.userAnonymisation];
693 if (sessionDict) {
694 [contexts addObject:[[SPSelfDescribingJson alloc] initWithSchema:kSPSessionContextSchema andData:sessionDict]];
695 } else {
696 SPLogTrack(nil, @"Unable to get session context for eventId: %@", eventId);
697 }
698 }
699
700 // Add GDPR context
701 if (self.gdpr) {
702 SPSelfDescribingJson *gdprContext = self.gdpr.context;
703 if (gdprContext) {
704 [contexts addObject:gdprContext];
705 }
706 }
707}

720- (void)wrapContexts:(NSArray<SPSelfDescribingJson *> *)contexts toPayload:(SPPayload *)payload {
721 if (contexts.count == 0) {
722 return;
723 }
724 NSMutableArray<NSDictionary *> *data = [NSMutableArray new];
725 for (SPSelfDescribingJson *context in contexts) {
726 [data addObject:[context getAsDictionary]];
727 }
728
729 SPSelfDescribingJson *finalContext = [[SPSelfDescribingJson alloc] initWithSchema:kSPContextSchema andData:data];
730 if (finalContext == nil) {
731 return;
732 }
733 [payload addDictionaryToPayload:[finalContext getAsDictionary]
734 base64Encoded:_base64Encoded
735 typeWhenEncoded:kSPContextEncoded
736 typeWhenNotEncoded:kSPContext];
737}
