96- (void)setupNetworkConnection {
97 if (!_builderFinished && _networkConnection) {
98 return;
99 }
100 __weak __typeof__(self) weakSelf = self;
101 _networkConnection = [SPDefaultNetworkConnection build:^(id<SPDefaultNetworkConnectionBuilder> builder) {
102 __typeof__(self) strongSelf = weakSelf;
103 if (!strongSelf) return;
104 NSString *endpoint = strongSelf->_url ? strongSelf->_url : @""; // use empty string in case nil to avoid crashing
105 if (![endpoint hasPrefix:@"http"]) {
106 NSString *protocol = strongSelf->_protocol == SPProtocolHttps ? @"https://" : @"http://";
107 endpoint = [protocol stringByAppendingString:endpoint];
108 }
109 [builder setUrlEndpoint:endpoint];
110 [builder setCustomPostPath:strongSelf->_customPostPath];
111 [builder setHttpMethod:strongSelf->_httpMethod];
112 [builder setRequestHeaders:strongSelf->_requestHeaders];
113 [builder setEmitThreadPoolSize:strongSelf->_emitThreadPoolSize];
114 [builder setByteLimitGet:strongSelf->_byteLimitGet];
115 [builder setByteLimitPost:strongSelf->_byteLimitPost];
116 [builder setServerAnonymisation:strongSelf->_serverAnonymisation];
117 }];
118}

293- (void) sendGuard {
294 if (_isSending || _pausedEmit) {
295 return;
296 }
297 @synchronized (self) {
298 if (!_isSending && !_pausedEmit) {
299 _isSending = YES;
300 @try {
301 [self attemptEmit];
302 } @catch (NSException *exception) {
303 SPLogError(@"Received exception during emission process: %@", exception);
304 _isSending = NO;
305 }
306 }
307 }

310- (void)attemptEmit {
311 if (!_eventStore.count) {
312 SPLogDebug(@"Database empty. Returning.", nil);
313 _isSending = NO;
314 return;
315 }
316
317 NSArray<SPEmitterEvent *> *events = [_eventStore emittableEventsWithQueryLimit:_emitRange];
318 NSArray<SPRequest *> *requests = [self buildRequestsFromEvents:events];
319 NSArray<SPRequestResult *> *sendResults = [_networkConnection sendRequests:requests];
320
321 SPLogVerbose(@"Processing emitter results.");
322
323 NSInteger successCount = 0;
324 NSInteger failedWillRetryCount = 0;
325 NSInteger failedWontRetryCount = 0;
326 NSMutableArray<NSNumber *> *removableEvents = [NSMutableArray new];
327
328 for (SPRequestResult *result in sendResults) {
329 NSArray<NSNumber *> *resultIndexArray = result.storeIds;
330 if (result.isSuccessful) {
331 successCount += resultIndexArray.count;
332 [removableEvents addObjectsFromArray:resultIndexArray];
333 } else if ([result shouldRetry:_customRetryForStatusCodes]) {
334 failedWillRetryCount += resultIndexArray.count;
335 } else {
336 failedWontRetryCount += resultIndexArray.count;
337 [removableEvents addObjectsFromArray:resultIndexArray];
338 SPLogError(@"Sending events to Collector failed with status %ld. Events will be dropped.", (long)[result statusCode]);
339 }
340 }
341 NSInteger allFailureCount = failedWillRetryCount + failedWontRetryCount;
342
343 [_eventStore removeEventsWithIds:removableEvents];
344
345 SPLogDebug(@"Success Count: %@", [@(successCount) stringValue]);
346 SPLogDebug(@"Failure Count: %@", [@(allFailureCount) stringValue]);
347
348 if (_callback != nil) {
349 if (allFailureCount == 0) {
350 [_callback onSuccessWithCount:successCount];
351 } else {
352 [_callback onFailureWithCount:allFailureCount successCount:successCount];
353 }
354 }
355
356 if (failedWillRetryCount > 0 && successCount == 0) {
357 SPLogDebug(@"Ending emitter run as all requests failed.", nil);
358 [NSThread sleepForTimeInterval:5];
359 _isSending = NO;
360 return;
361 } else {
362 [self attemptEmit];
363 }
364}

366- (NSArray<SPRequest *> *)buildRequestsFromEvents:(NSArray<SPEmitterEvent *> *)events {
367 NSMutableArray<SPRequest *> *requests = [NSMutableArray new];
368 NSNumber *sendingTime = [SPUtilities getTimestamp];
369 SPHttpMethod httpMethod = _networkConnection.httpMethod;
370
371 if (httpMethod == SPHttpMethodGet) {
372 for (SPEmitterEvent *event in events) {
373 SPPayload *payload = event.payload;
374 [self addSendingTimeToPayload:payload timestamp:sendingTime];
375 BOOL oversize = [self isOversize:payload];
376 SPRequest *request = [[SPRequest alloc] initWithPayload:payload emitterEventId:event.storeId oversize:oversize];
377 [requests addObject:request];
378 }
379 } else {
380 for (int i = 0; i < events.count; i += _bufferOption) {
381 NSMutableArray<SPPayload *> *eventArray = [NSMutableArray new];
382 NSMutableArray<NSNumber *> *indexArray = [NSMutableArray new];
383
384 for (int j = i; j < (i + _bufferOption) && j < events.count; j++) {
385 SPEmitterEvent *event = events[j];
386
387 SPPayload *payload = event.payload;
388 NSNumber *emitterEventId = @(event.storeId);
389 [self addSendingTimeToPayload:payload timestamp:sendingTime];
390
391 if ([self isOversize:payload]) {
392 SPRequest *request = [[SPRequest alloc] initWithPayload:payload emitterEventId:emitterEventId.longLongValue oversize:YES];
393 [requests addObject:request];
394
395 } else if ([self isOversize:payload previousPayloads:eventArray]) {
396 SPRequest *request = [[SPRequest alloc] initWithPayloads:eventArray emitterEventIds:indexArray];
397 [requests addObject:request];
398
399 // Clear collection and build a new POST
400 eventArray = [NSMutableArray new];
401 indexArray = [NSMutableArray new];
402
403 // Build and store the request
404 [eventArray addObject:payload];
405 [indexArray addObject:emitterEventId];
406
407 } else {
408 // Add event to collections
409 [eventArray addObject:payload];
410 [indexArray addObject:emitterEventId];
411 }
412 }
413
414 // Check if all payloads have been processed
415 if (eventArray.count) {
416 SPRequest *request = [[SPRequest alloc] initWithPayloads:eventArray emitterEventIds:indexArray];
417 [requests addObject:request];
418 }
419 }
420 }
421 return requests;
422}

428- (BOOL)isOversize:(SPPayload *)payload previousPayloads:(NSArray<SPPayload *> *)previousPayloads {
429 NSUInteger byteLimit = _networkConnection.httpMethod == SPHttpMethodGet ? _byteLimitGet : _byteLimitPost;
430 return [self isOversize:payload byteLimit:byteLimit previousPayloads:previousPayloads];

433- (BOOL)isOversize:(SPPayload *)payload byteLimit:(NSUInteger)byteLimit previousPayloads:(NSArray<SPPayload *> *)previousPayloads {
434 NSUInteger totalByteSize = payload.byteSize;
435 for (SPPayload *previousPayload in previousPayloads) {
436 totalByteSize += previousPayload.byteSize;
437 }
438 NSUInteger wrapperBytes = previousPayloads.count > 0 ? (previousPayloads.count + POST_WRAPPER_BYTES) : 0;
439 return totalByteSize + wrapperBytes > byteLimit;
440}
