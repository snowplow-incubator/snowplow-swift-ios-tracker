43 if (self = [super init]) {

57- (void)retrieveConfigurationOnlyRemote:(BOOL)onlyRemote onFetchCallback:(OnFetchCallback)onFetchCallback {
58 @synchronized (self) {
59 if (!onlyRemote) {
60 if (!self.cacheBundle) {
61 self.cacheBundle = [self.cache readCache];
62 }
63 if (self.cacheBundle) {
64 onFetchCallback(self.cacheBundle, SPConfigurationStateCached);
65 } else if (self.defaultBundle) {
66 onFetchCallback(self.defaultBundle, SPConfigurationStateDefault);
67 }
68 }
69 self.fetcher = [[SPConfigurationFetcher alloc] initWithRemoteSource:self.remoteConfiguration onFetchCallback:^(SPFetchedConfigurationBundle * _Nonnull fetchedConfigurationBundle, SPConfigurationState configurationState) {
70 if (![self schemaCompatibility:fetchedConfigurationBundle.schema]) {
71 return;
72 }
73 @synchronized (self) {
74 if (self.cacheBundle && self.cacheBundle.configurationVersion >= fetchedConfigurationBundle.configurationVersion) {
75 return;
76 }
77 [self.cache writeCache:fetchedConfigurationBundle];
78 self.cacheBundle = fetchedConfigurationBundle;
79 onFetchCallback(fetchedConfigurationBundle, SPConfigurationStateFetched);
80 }
81 }];
82 }
