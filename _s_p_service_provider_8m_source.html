91- (instancetype)initWithNamespace:(NSString *)namespace network:(SPNetworkConfiguration *)networkConfiguration configurations:(NSArray<SPConfiguration *> *)configurations {
92 if (self = [super init]) {
93 [self initializeConfigurationUpdates];
94 self.namespace = namespace;
95 self.networkConfigurationUpdate.sourceConfig = networkConfiguration;
96 [self processConfigurations:configurations];
97 if (!self.trackerConfigurationUpdate.sourceConfig) {
98 self.trackerConfigurationUpdate.sourceConfig = [SPTrackerConfiguration new];
99 }
100 [self tracker]; // Build tracker to initialize NotificationCenter receivers
101 }
102 return self;
103}

123- (void)processConfigurations:(NSArray<SPConfiguration *> *)configurations {
124 for (SPConfiguration *configuration in configurations) {
125 if ([configuration isKindOfClass:SPNetworkConfiguration.class]) {
126 self.networkConfigurationUpdate.sourceConfig = (SPNetworkConfiguration *)configuration;
127 continue;
128 }
129 if ([configuration isKindOfClass:SPTrackerConfiguration.class]) {
130 self.trackerConfigurationUpdate.sourceConfig = (SPTrackerConfiguration *)configuration;
131 continue;
132 }
133 if ([configuration isKindOfClass:SPSubjectConfiguration.class]) {
134 self.subjectConfigurationUpdate.sourceConfig = (SPSubjectConfiguration *)configuration;
135 continue;
136 }
137 if ([configuration isKindOfClass:SPSessionConfiguration.class]) {
138 self.sessionConfigurationUpdate.sourceConfig = (SPSessionConfiguration *)configuration;
139 continue;
140 }
141 if ([configuration isKindOfClass:SPEmitterConfiguration.class]) {
142 self.emitterConfigurationUpdate.sourceConfig = (SPEmitterConfiguration *)configuration;
143 continue;
144 }
145 if ([configuration isKindOfClass:SPGDPRConfiguration.class]) {
146 self.gdprConfigurationUpdate.sourceConfig = (SPGDPRConfiguration *)configuration;
147 continue;
148 }
149 if ([configuration isKindOfClass:SPGlobalContextsConfiguration.class]) {
150 self.globalContextConfiguration = (SPGlobalContextsConfiguration *)configuration;
151 continue;
152 }
153 }

272- (SPEmitter *)makeEmitter {
273 SPNetworkConfigurationUpdate *networkConfig = self.networkConfigurationUpdate;
274 SPEmitterConfigurationUpdate *emitterConfig = self.emitterConfigurationUpdate;
275 SPEmitter *emitter = [SPEmitter build:^(id<SPEmitterBuilder> builder) {
276 if (networkConfig.networkConnection) {
277 [builder setNetworkConnection:networkConfig.networkConnection];
278 } else {
279 [builder setHttpMethod:networkConfig.method];
280 [builder setProtocol:networkConfig.protocol];
281 [builder setUrlEndpoint:networkConfig.endpoint];
282 }
283 [builder setCustomPostPath:networkConfig.customPostPath];
284 [builder setRequestHeaders:networkConfig.requestHeaders];
285 if (emitterConfig) {
286 [builder setEmitRange:emitterConfig.emitRange];
287 [builder setBufferOption:emitterConfig.bufferOption];
288 [builder setEventStore:emitterConfig.eventStore];
289 [builder setByteLimitPost:emitterConfig.byteLimitPost];
290 [builder setByteLimitGet:emitterConfig.byteLimitGet];
291 [builder setEmitThreadPoolSize:emitterConfig.threadPoolSize];
292 [builder setCallback:emitterConfig.requestCallback];
293 [builder setCustomRetryForStatusCodes:emitterConfig.customRetryForStatusCodes];
294 [builder setServerAnonymisation:emitterConfig.serverAnonymisation];
295 }
296 }];
297 if (emitterConfig && emitterConfig.isPaused) {
298 [emitter pauseEmit];
299 }
300 return emitter;
301}

303- (SPTracker *)makeTracker {
304 SPEmitter *emitter = self.emitter;
305 SPSubject *subject = self.subject;
306 SPTrackerConfiguration *trackerConfig = self.trackerConfigurationUpdate;
307 SPSessionConfiguration *sessionConfig = self.sessionConfigurationUpdate;
308 SPGlobalContextsConfiguration *gcConfig = self.globalContextConfiguration;
309 SPTracker *tracker = [SPTracker build:^(id<SPTrackerBuilder> builder) {
310 [builder setTrackerNamespace:self.namespace];
311 [builder setEmitter:emitter];
312 [builder setSubject:subject];
313 [builder setAppId:trackerConfig.appId];
314 [builder setTrackerVersionSuffix:trackerConfig.trackerVersionSuffix];
315 [builder setBase64Encoded:trackerConfig.base64Encoding];
316 [builder setLogLevel:trackerConfig.logLevel];
317 [builder setLoggerDelegate:trackerConfig.loggerDelegate];
318 [builder setDevicePlatform:trackerConfig.devicePlatform];
319 [builder setSessionContext:trackerConfig.sessionContext];
320 [builder setApplicationContext:trackerConfig.applicationContext];
321 [builder setDeepLinkContext:trackerConfig.deepLinkContext];
322 [builder setScreenContext:trackerConfig.screenContext];
323 [builder setAutotrackScreenViews:trackerConfig.screenViewAutotracking];
324 [builder setLifecycleEvents:trackerConfig.lifecycleAutotracking];
325 [builder setInstallEvent:trackerConfig.installAutotracking];
326 [builder setExceptionEvents:trackerConfig.exceptionAutotracking];
327 [builder setTrackerDiagnostic:trackerConfig.diagnosticAutotracking];
328 [builder setUserAnonymisation:trackerConfig.userAnonymisation];
329 if (sessionConfig) {
330 [builder setBackgroundTimeout:sessionConfig.backgroundTimeoutInSeconds];
331 [builder setForegroundTimeout:sessionConfig.foregroundTimeoutInSeconds];
332 }
333 if (gcConfig) {
334 [builder setGlobalContextGenerators:gcConfig.contextGenerators];
335 }
336 SPGDPRConfigurationUpdate *gdprConfig = self.gdprConfigurationUpdate;
337 if (gdprConfig.sourceConfig) {
338 [builder setGdprContextWithBasis:gdprConfig.basisForProcessing documentId:gdprConfig.documentId documentVersion:gdprConfig.documentVersion documentDescription:gdprConfig.documentDescription];
339 }
340 }];
341 if (self.trackerConfigurationUpdate.isPaused) {
342 [tracker pauseEventTracking];
343 }
344 if (tracker.session) {
345 if (self.sessionConfigurationUpdate.isPaused) {
346 [tracker.session stopChecker];
347 }
348 if (self.sessionConfigurationUpdate.onSessionStateUpdate) {
349 tracker.session.onSessionStateUpdate = self.sessionConfigurationUpdate.onSessionStateUpdate;
350 }
351 }
352 return tracker;
353}
