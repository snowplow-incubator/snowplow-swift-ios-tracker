61- (SPPayload *) fetchPlatformDictWithUserAnonymisation:(BOOL)userAnonymisation {
62#if SNOWPLOW_TARGET_IOS
63 @synchronized (self) {
64 NSTimeInterval now = [[NSDate date] timeIntervalSince1970];
65 if (now - self.lastUpdatedEphemeralMobileDict >= self.mobileDictUpdateFrequency) {
66 [self setEphemeralMobileDict];
67 }
68 if (now - self.lastUpdatedEphemeralNetworkDict >= self.networkDictUpdateFrequency) {
69 [self setEphemeralNetworkDict];
70 }
71 }
72#endif
73 if (userAnonymisation) { // mask user identifiers
74 SPPayload *copy = [[SPPayload alloc] initWithNSDictionary:[self.platformDict getAsDictionary]];
75 [copy addValueToPayload:nil forKey:kSPMobileAppleIdfa];
76 [copy addValueToPayload:nil forKey:kSPMobileAppleIdfv];
77 return copy;
78 } else {
79 return self.platformDict;
80 }

106- (void) setEphemeralMobileDict {
107 self.lastUpdatedEphemeralMobileDict = [[NSDate date] timeIntervalSince1970];
108
109 NSDictionary *currentDict = [self.platformDict getAsDictionary];
110 if ([currentDict valueForKey:kSPMobileAppleIdfa] == nil) {
111 [self.platformDict addValueToPayload:[self.deviceInfoMonitor appleIdfa] forKey:kSPMobileAppleIdfa];
112 }
113 if ([currentDict valueForKey:kSPMobileAppleIdfv] == nil) {
114 [self.platformDict addValueToPayload:[self.deviceInfoMonitor appleIdfv] forKey:kSPMobileAppleIdfv];
115 }
116
117 [self.platformDict addNumericValueToPayload:[self.deviceInfoMonitor batteryLevel] forKey:kSPMobileBatteryLevel];
118 [self.platformDict addValueToPayload:[self.deviceInfoMonitor batteryState] forKey:kSPMobileBatteryState];
119 [self.platformDict addNumericValueToPayload:[self.deviceInfoMonitor isLowPowerModeEnabled] forKey:kSPMobileLowPowerMode];
120 [self.platformDict addNumericValueToPayload:[self.deviceInfoMonitor availableStorage] forKey:kSPMobileAvailableStorage];
121 [self.platformDict addNumericValueToPayload:[self.deviceInfoMonitor appAvailableMemory] forKey:kSPMobileAppAvailableMemory];
122}

128 [self.platformDict addValueToPayload:[self.deviceInfoMonitor networkType] forKey:kSPMobileNetworkType];
