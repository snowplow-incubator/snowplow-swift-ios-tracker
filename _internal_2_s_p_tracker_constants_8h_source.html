24// Macros to define what OS is running:
25// 1. iOS: iOS == 1; OSX == 1; tvOS == 0 ; watchOS == 0
26// 2. OSX: iOS == 0; OSX == 1; tvOS == 0
27// 3. TV: iOS == 1; OSX == 1; tvOS == 1
28#define SNOWPLOW_TARGET_IOS (TARGET_OS_IPHONE && TARGET_OS_MAC && !(TARGET_OS_TV) && !(TARGET_OS_WATCH))
29#define SNOWPLOW_TARGET_OSX (!(TARGET_OS_IPHONE) && TARGET_OS_MAC && !(TARGET_OS_TV))
30#define SNOWPLOW_TARGET_TV (TARGET_OS_IPHONE && TARGET_OS_MAC && TARGET_OS_TV)
31#define SNOWPLOW_TARGET_WATCHOS (TARGET_OS_WATCH)

40// Macros for builder pattern
41#define SP_ESCAPE(...) __VA_ARGS__
42#define SP_BUILDER_DECLARE(type, prop) - (instancetype)prop:(type)value NS_SWIFT_NAME(prop(_:));
43#define SP_BUILDER_DECLARE_NULLABLE(type, prop) - (instancetype)prop:(nullable type)value NS_SWIFT_NAME(prop(_:));
44#define SP_BUILDER_METHOD(type, prop) - (instancetype)prop:(type)value { self.prop = value; return self; }
