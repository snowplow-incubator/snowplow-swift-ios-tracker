41- (id<SPState>)transitionFromEvent:(SPEvent *)event state:(id<SPState>)currentState {
42 SPScreenView *screenView = (SPScreenView *)event;
43 SPScreenState *oldState = (SPScreenState *)currentState;
44 SPScreenState *newState = [self screenStateFromScreenView:screenView];
45 newState.previousState = oldState;
46 return newState;
47}

49- (NSArray<SPSelfDescribingJson *> *)entitiesFromEvent:(id<SPInspectableEvent>)event state:(id<SPState>)state {
50 if ([state isKindOfClass:SPScreenState.class]) {
51 SPSelfDescribingJson *entity = [self screenContextFromScreenState:(SPScreenState *)state];
52 return @[entity];
53 }
54 return nil;
55}

57- (NSDictionary<NSString *,NSObject *> *)payloadValuesFromEvent:(id<SPInspectableEvent>)event state:(id<SPState>)state {
58 if ([state isKindOfClass:SPScreenState.class]) {
59 SPScreenState *previousState = ((SPScreenState *)state).previousState;
60 NSMutableDictionary<NSString *,NSObject *> *addedValues = [NSMutableDictionary new];
61 [addedValues setValue:previousState.name forKey:kSPSvPreviousName];
62 [addedValues setValue:previousState.type forKey:kSPSvPreviousType];
63 [addedValues setValue:previousState.screenId forKey:kSPSvPreviousScreenId];
64 return addedValues;
65 }
66 return nil;
67}

71- (SPScreenState *)screenStateFromScreenView:(SPScreenView *)screenView {
72 return [[SPScreenState alloc] initWithName:screenView.name

80- (SPSelfDescribingJson *)screenContextFromScreenState:(SPScreenState *)screenState {
81 SPPayload *contextPayload = [screenState payload];
82 if (!contextPayload) {
83 return nil;
84 }
85 return [[SPSelfDescribingJson alloc] initWithSchema:kSPScreenContextSchema andPayload:contextPayload];
