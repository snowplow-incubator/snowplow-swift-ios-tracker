75- (instancetype)initWithStoredState:(NSDictionary<NSString *,NSObject *> *)storedState {
76 if (self = [super init]) {
77 self.sessionId = [storedState sp_stringForKey:kSPSessionId defaultValue:nil];
78 if (!self.sessionId) return nil;
79
80 NSNumber *sessionIndexNumber = [storedState sp_numberForKey:kSPSessionIndex defaultValue:nil];
81 if (!sessionIndexNumber) return nil;
82 self.sessionIndex = sessionIndexNumber.integerValue;
83
84 self.userId = [storedState sp_stringForKey:kSPSessionUserId defaultValue:nil];
85 if (!self.userId) return nil;
86
87 self.previousSessionId = [storedState sp_stringForKey:kSPSessionPreviousId defaultValue:nil];
88
89 // The FirstEventId should be stored in legacy persisted sessions even
90 // if it wasn't used. Anyway we provide a default value in order to be
91 // defensive and exclude any possible issue with a missing value.
92 self.firstEventId = [storedState sp_stringForKey:kSPSessionFirstEventId
93 defaultValue:@"00000000-0000-0000-0000-000000000000"];
94 self.firstEventTimestamp = [storedState sp_stringForKey:kSPSessionFirstEventTimestamp defaultValue:nil];
95
96 self.storage = [storedState sp_stringForKey:kSPSessionStorage defaultValue:@"LOCAL_STORAGE"];
97
98 self.sessionDictionary = [SPSessionState buildSessionDictionaryWithFirstEventId:self.firstEventId
99 firstEventTimestamp:self.firstEventTimestamp
100 currentSessionId:self.sessionId
101 previousSessionId:self.previousSessionId
102 sessionIndex:self.sessionIndex
103 userId:self.userId
104 storage:self.storage];
105 }
